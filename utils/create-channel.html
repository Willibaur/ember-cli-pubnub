import Ember from 'ember'; import Channel from 'ember-cli-pubnub/channels/channel'; const { assert } = Ember;

Method createChannel

Create a channel object with a given name. This will be the interface to the PubNub service when interacting with this channel.

Parameters:

  • name must be a String.
    (The name of the channel.)

  • service must be an Ember.Service.
    (The pubnub service.)

Returns an Ember.Object
(A channel object.)

export default function createChannel(name, service) { assert('You must give the channel a name', name); const channel = Channel.create({ name: name, pubnub: service }); return channel; }
