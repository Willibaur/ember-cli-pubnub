addon/models/channel.js
import Ember from 'ember'; const { get } = Ember; export default Ember.Object.extend({

Method audit

Reveal existing permission attributes set on this channel.

Parameters:

  • callback must be a Function.
    (Callback method that will return the permission attributes.)
audit(callback) { const channel = get(this, 'name'); const pubnub = get(this, 'pubnub'); pubnub.audit(channel, callback); },

Method grant

Set permissions on this channel.

Parameters:

  • options must be an Object.
    (Permissions to set on the channel.)
grant(options) { const channel = get(this, 'name'); const pubnub = get(this, 'pubnub'); pubnub.grant(channel, options); },

Method hereNow

Obtain information about the current state of this channel including a list of unique UUIDs currently subscribed to a channel and the total occupancy count.

hereNow() { const channel = get(this, 'name'); const pubnub = get(this, 'pubnub'); const presence = pubnub.listPresence(channel); if (!presence) { return pubnub.hereNow(channel); } return presence; },

Method history

Fetch historical messages from this channel. The results of running the history will be returned by the registered event handler for the channel.

Parameters:

  • options must be an Object.
    (Options to pass the history method.)
history(options) { const channel = get(this, 'name'); const pubnub = get(this, 'pubnub'); pubnub.history(channel, options); },

Method on

Attach an event to the channel attaching a callback for when is it fired.

Parameters:

  • event must be a String.
    (The event name or a function that will return an event string.)

  • callback must be a Function.
    (A function that will be fired on the event.)

on(event, callback) { const channel = get(this, 'name'); const pubnub = get(this, 'pubnub');

Grab an event string if the event is a function.

if (typeof event === 'function') { event = event.call(this, channel); } pubnub.on(event, callback); },

Method publish

Publish messages to this channel.

Parameters:

  • message must be an Object.
    (Any valid JSON type including objects, arrays, strings, and numbers.)
publish(message) { const channel = get(this, 'name'); const pubnub = get(this, 'pubnub'); pubnub.publish(channel, message); },

Method revoke

Remove any non default permissions from a channel.

revoke() { const channel = get(this, 'name'); const pubnub = get(this, 'pubnub'); pubnub.revoke(channel); },

Method unsubscribe

Unsubscribe from this channel.

unsubscribe() { const channel = get(this, 'name'); const pubnub = get(this, 'pubnub'); pubnub.unsubscribe(channel); } });
